# Tools
find -> find file in su

trap -> trapping signals
	types of signals (SIGINT, SIGKILL, etc) 
	traps a specific signal and perform a predefined task
	trap "echo don't use the ctrl+c" SIGINT


kill -> end a specific process
	ex) kill -9 1111
